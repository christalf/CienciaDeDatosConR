# Using the sqldf package to manipulate R data frames with SQL commands.
# Remember:
# in order to use `sqldf` package we need to load only that package, it doesn’t
# need RMySQL or any other like that; if both sqldf and RMySQL are loaded, there
# could be problems so we should better deattach RMySQL.

if("RMySQL" %in% (.packages())) {
    detach("package:RMySQL", unload = TRUE)
}

3 Segundo ejercicio - Github API

Acceso a la API de Github con el paquete httr.

# GitHub API access through "httr" R package (OAuth protocol)

library(httpuv)
library(httr)
library(jsonlite)

# This function provides some common OAuth endpoints (linkedin, twitter,
# vimeo, google, facebook, github, azure).
oauth_endpoints("github")
<oauth_endpoint>
 authorize: https://github.com/login/oauth/authorize
 access:    https://github.com/login/oauth/access_token

# Create an OAuth app.
myapp <- oauth_app("github",
                   key = Sys.getenv("GITHUB_API_KEY"),
                   secret = Sys.getenv("GITHUB_API_SECRET_KEY"))

# Get OAuth credentials.
# - Esta instrucción abre una página web donde hay que autenticarse para
#   tener acceso a la API de Github.
github_token <- oauth2.0_token(oauth_endpoints("github"), myapp)

# Use the API.

4 Tercer ejercicio - Files de formato fijo

Lectura de files con formato fijo con el paquete de base utils.

# Reading fixed-format files.

# "utils" R package has functions to read fixed-format files.
# For example, the "read.fortran()" function to read fortran files (".for" extension).
# Or the "read.fwf()" function, to read in general Fixed Width Format files (we'll use
# this one).
# Both functions returns a "data.frame" object (they internally call read.table() function).

# Using "read.fwf()" function.
# It's first argument can be the name of the file or a connection (which will be opened if
# necessary, and if so closed at the end of the function call).
str(read.fwf)
function (file, widths, header = FALSE, sep = "\t", skip = 0L, row.names, 
    col.names, n = -1L, buffersize = 2000, fileEncoding = "", ...)  

# So, we create a connection to the file (it's on a webserver) and let's first read

# columns (negative numbers) and the number of initial lines to skip (all this, by looking
# thoroughly at the file on the web).
